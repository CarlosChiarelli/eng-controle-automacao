t = np.arange(0, 40, 1e-2) # tempo (0 a 20s)

def resposta_funcao_Csi(csi, t):

    Gs_num = omega_n**2
    Gs_den = [1, 2*Csi*omega_n, omega_n**2]

    Gs = (Gs_num, Gs_den)

    LTI = lti(Gs_num, Gs_den) # gerando o sistema

    t_step, y_step = step(LTI, T=t)      # resposta ao impulso
    
    return t_step, y_step

Modelo mecnico:

$$ $$$$ T(t) = J \ddot \theta(t) + B \dot \theta(t) + K \theta(t) $$$$ $$$$ \omega_{n} = \sqrt \frac{K}{J} $$$$ $$$$ \xi = \frac{B}{2 \sqrt {KJ}} $$$$ $$$$ \omega_{d} = \omega_{n} \sqrt{1 - \xi ^2} $$

def find_nearest(a, a0):
    "Encontra o elemento mais prximo dentro o array a com valor a0"
    idx = np.abs(a - a0).argmin()
    return a.flat[idx]
